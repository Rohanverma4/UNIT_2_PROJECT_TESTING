javaScript Code -- Location Change



var form = document.querySelector("form");
console.log(form);
form.addEventListener("submit",function(event){

	event.preventDefault();


	var locationName = document.getElementsByName("location");
	
	for(var i = 0 ; i < locationName.length ; i++){
		if(locationName[i].checked){
			document.getElementById("locationBox").innerHTML = locationName[i].value;
		}
	}
	
})
